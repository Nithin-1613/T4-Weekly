int kthSmallest(int arr[], int l, int r, int k) {
        //code here
        priority_queue<int> g;
        for(int i=l;i<=r;i++)
        {
            g.push(arr[i]);
        }
        k=r-k+1;
        while(k>0)
        {
            g.pop();
            k--;
        }
        return g.top();
    }
