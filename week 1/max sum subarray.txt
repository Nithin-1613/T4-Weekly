class Solution {
public:
    // KADANE ALGORITHM
    int maxSubArray(vector<int>& nums) {
        int max_sum=nums[0],curr_sum=0;
        for(int i=0;i<nums.size();i++)
        {
            curr_sum+=nums[i];
            if(curr_sum<nums[i])
                curr_sum=nums[i];
            max_sum=max(max_sum,curr_sum);
        }
        return max_sum;
    }
};
