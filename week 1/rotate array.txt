class Solution {
public:
    void reverse(vector<int>& nums,int a,int b)
    {
        for(int i=a,j=b;i<j;i++,j--)
        {
            int t=nums[i];
            nums[i]=nums[j];
            nums[j]=t;
        }
    }
    void rotate(vector<int>& nums, int k) {
        
        int n=nums.size();
        if(k<0 )
            return;
        while(k>=n)
            k-=n;
        reverse(nums,n-k,n-1);
        reverse(nums,0,n-k-1);
        reverse(nums,0,n-1);
        
    }
};
