int majorityElement(int a[], int size)
    {
        
        // your code here
        map<int,int> A;
        for(int i=0;i<size;i++)
        {
            A[a[i]]++;
        }
        int num=-1;
        for(auto j=A.begin();j!=A.end();j++)
        {
            if(j->second>(size/2))
                num=j->first;
        }
     return num;   
    }
