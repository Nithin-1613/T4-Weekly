class Solution{
    public:
    vector<pair<int,int>> allPairs(int A[], int B[], int N, int M, int X)
    {
        // Your code goes here 
        vector<pair<int,int>> res;
        sort(A,A+N);
        for(int i=0;i<N;i++)
        {
            for(int j=0;j<M;j++)
            {
                if(A[i]+B[j]==X)
                    res.push_back({A[i],B[j]});
            }
        }
    
        return res;
    }
};
