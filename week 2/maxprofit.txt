class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n=prices.size();
        //int profit=0;
        int buy=INT_MAX,sell=0;
        for(int i=0;i<n;i++)
        {
            if(prices[i]<buy)
                buy=prices[i];
            else if(prices[i]-buy>sell)
                sell=prices[i]-buy;
        }
        return sell;
    }
};
